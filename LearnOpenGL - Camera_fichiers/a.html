                  Hello! I would suggest an improvement I came up with due to the limited float precision. If you keep rotating on the Y axis (yaw), your yaw values will get very, very high! To fix this:

                  #include <math.h>
                  ...
                  // If we don't constrain the yaw to only use values between 0-360
                  // we would lose floating precission with very high values, hence
                  // the movement would look like big "steps" instead a smooth one!
                  Yaw = std::fmod((Yaw + xoffset), (GLfloat)360.0f);

                            Laurie Bradshaw 3 years ago

                            Just thought I'd share an example glm free camera, as tutorials, this one included, all seem to demonstrate fps cameras.

                            class camera {
                            	glm::vec3 m_pos;
                            	glm::quat m_orient;
                            public:
                            	camera(void) = default;
                            	camera(const camera &) = default;

                            	camera(const glm::vec3 &pos) : m_pos(pos) {}
                            	camera(const glm::vec3 &pos, const glm::quat &orient) : m_pos(pos), m_orient(orient) {}

                            	camera &operator =(const camera &) = default;

                            	const glm::vec3 &position(void) const { return m_pos; }
                            	const glm::quat &orientation(void) const { return m_orient; }

                            	glm::mat4 view(void) const { return glm::translate(glm::mat4_cast(m_orient), m_pos); }

                            	void translate(const glm::vec3 &v) { m_pos += v * m_orient; }
                            	void translate(float x, float y, float z) { m_pos += glm::vec3(x, y, z) * m_orient; }

                            	void rotate(float angle, const glm::vec3 &axis) { m_orient *= glm::angleAxis(angle, axis * m_orient); }
                            	void rotate(float angle, float x, float y, float z) { m_orient *= glm::angleAxis(angle, glm::vec3(x, y, z) * m_orient); }

                            	void yaw(float angle) { rotate(angle, 0.0f, 1.0f, 0.0f); }
                            	void pitch(float angle) { rotate(angle, 1.0f, 0.0f, 0.0f); }
                            	void roll(float angle) { rotate(angle, 0.0f, 0.0f, 1.0f); }
                            };

                                Here is how you can calculate your orientation vectors at runtime without worrying about roll, pitch, yaw, or calling sin, cos

                                // orientation is a quaternion and I Front, Up, and Right are glm::vec3

                                void QuatCam::updateCameraVectors()
                                {
                                Front = glm::vec3(
                                2.0f * (Orientation.x * Orientation.z + Orientation.w * Orientation.y),
                                2.0f * (Orientation.y * Orientation.z - Orientation.w * Orientation.x),
                                1.0f - 2.0f * (Orientation.x * Orientation.x + Orientation.y * Orientation.y)
                                );

                                Up = glm::vec3(
                                2.0f * (Orientation.x * Orientation.y - Orientation.w * Orientation.z),
                                1.0f - 2.0f * (Orientation.x * Orientation.x + Orientation.z * Orientation.z),
                                2.0f * (Orientation.y * Orientation.z + Orientation.w * Orientation.x)
                                );

                                Right = glm::vec3(
                                -(1.0f - 2.0f * (Orientation.y * Orientation.y + Orientation.z * Orientation.z)),
                                -(2.0f * (Orientation.x * Orientation.y + Orientation.w * Orientation.z)),
                                -(2.0f * (Orientation.x * Orientation.z - Orientation.w * Orientation.y))
                                );

                                This is useful if you want to improve performance or want to perform a rotation or translation along these axis.
                                Im not sure why Laurie modifies the orientation during a translation but this has worked just fine for me.

                                void QuatCam::translate(const glm::vec3 &v)
                                {
                                Position += v;
                                }

                                void QuatCam::rotate(float angle, const glm::vec3 &axis)
                                {
                                Orientation *= glm::angleAxis(angle, axis * Orientation);
                                updateCameraVectors(); // update vectors everytime an orientation changes
                                }

                                    CHAN HAENG Lee Tarrant Mangasarian a year ago

                                    your code in updateCameraVectors() is retrieving the direction from a quaternion. But, I found the errors on the signs of operations, compared to other articles and functions

                                    https://www.euclideanspace....
                                    in the above link, you can know the directions from a quaterion.

                                    and also in glm, there is a method converting from quarternion to rotation matrix; glm::mat4_cast()
                                    Actually the directions are in the rotation matrix.

                                    the internal implementation of the method is :
                                    (actually mat4_cast uses mat3_cast because quat-to-rotationMatrix needs 3x3 upper left matrix.)
                                    template<typename t,="" qualifier="" q="">
                                    GLM_FUNC_QUALIFIER mat<3, 3, T, Q> mat3_cast(tquat<t, q=""> const& q)
                                    {
                                    mat<3, 3, T, Q> Result(T(1));
                                    T qxx(q.x * q.x);
                                    T qyy(q.y * q.y);
                                    T qzz(q.z * q.z);
                                    T qxz(q.x * q.z);
                                    T qxy(q.x * q.y);
                                    T qyz(q.y * q.z);
                                    T qwx(q.w * q.x);
                                    T qwy(q.w * q.y);
                                    T qwz(q.w * q.z);

                                    Result[0][0] = T(1) - T(2) * (qyy + qzz); // Right.x
                                    Result[0][1] = T(2) * (qxy + qwz); // Up.x
                                    Result[0][2] = T(2) * (qxz - qwy); // Front.x

                                    Result[1][0] = T(2) * (qxy - qwz); // Right.y
                                    Result[1][1] = T(1) - T(2) * (qxx + qzz); // Up.y
                                    Result[1][2] = T(2) * (qyz + qwx); // Front.y

                                    Result[2][0] = T(2) * (qxz + qwy); // Right.z
                                    Result[2][1] = T(2) * (qyz - qwx); // Up.z
                                    Result[2][2] = T(1) - T(2) * (qxx + qyy); // Front.z
                                    return Result;
                                    }

                                    I commented on the code of matrix cast method. and you can see the difference between Tarrant code and those codes.
